Semantic

simple declarative binding DSL


                    val kodein = Kodein {
                        bind<DB>() with singleton { DBImpl() } // Infix functions!
                    }
Sentence: Bind the type DB to a Singleton containing a DB Implementation

Simple bindings


                    // Instance:
                    bind<DB>() with instance(db)

                    // Provider:
                    bind<Random>() with provider { SecureRandom() }

                    // Singleton:
                    bind<Net>() with singleton { NetImpl() }
Lambda!

Factory bindings


                    // Factory:
                    bind<Dice>() with factory { sides: Int -> DiceImpl(sides) }

                    // Multiton:
                    bind<Prefs>() with multiton { root: String -> PrefsImpl(root) }

Transitive dependencies


                    class Repo(val db: DB, val infos: AppInfos)
                

                    val kodein = Kodein {
                      bind<DB>() with singleton { DBImpl() }
                      bind<appInfos>() with instance(infos)

                      bind() from singleton { Repo(instance(), instance()) }
                    }

Retrieval


                    class MyManager(val kodein: Kodein) {
                        // Instance:
                        val db: DB = kodein.instance()

                        // Provider:
                        val getRandom: () -> Random = kodein.provider()

                        // Factory:
                        val getPrefs: (String) -> Prefs = kodein.factory()

                        // Currying:
                        val appPrefs: Prefs = kodein.with("app").instance()
                    }

                    interface KodeinAware {

KodeinAware


                    class MyManager(override val kodein: Kodein) : KodeinAware {
                        // Instance:
                        val db: DB = kodein.instance()

                        // Provider:
                        val getRandom: () -> Random = kodein.provider()

                        // Factory:
                        val getPrefs: (String) -> Prefs = kodein.factory()

                        // Currying:
                        val appPrefs: Prefs = kodein.with("app").instance()
                    }
